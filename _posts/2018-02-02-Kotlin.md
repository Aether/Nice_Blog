### Override

        open var a = 9;
        open fun v() {
            print("Base...")
        }
        fun nv() {}
    }
    class Subclass() : Base() {
        override var a = 8;//override attribute
        override fun v() {
            //super.v() call the method in baseclass
            print("Subclass...")//override method
        }
    }
